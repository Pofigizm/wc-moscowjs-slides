Веб компоненты

Спецификации

  • HTML Import - внешний html документ
  • <template> - пассивный DOM фрагмент (заготовка)
  • Custom elements - собственный элемент (тег)
  • Shadow DOM - скрытое дерево элемента

HTML Import

			<link rel="import" href="document.html">

До <template>

			<div hidden id="temp">
			    <img class="pict" src="picture.jpg">
			</div>
		
			<script tape="text/template">
			    <img class="pict" src="picture.jpg">
			</script>

<template>

			<template id="temp">
			    <img class="pict" src="picture.jpg">
			</template>
		
			var temp = document.querySelector('#temp');
			var clone = temp.content.cloneNode(true);
			document.body.appendChild(clone);

Особенности Custom elements

  • Собственная логика
  • Доступны события жизненого цикла
    • Создание элемента
    • Вставка в документ
    • Удаление из документа
    • Изменение атрибутов элемента
  • Имя элемента должно содержать дефис

Custom elements

			var proto = Object.create(HTMLElement.prototype);
			

				proto.createdCallback = function() {
				    this.innerHTML = 'Содержимое моего элемента.';
				};
			

			document.registerElement('my-elem', {prototype: proto});
		
			<my-elem></my-elem>

Shadow DOM

Shadow DOM

			<button id="butt">Светлая сторона</button>
			
		
			var host = document.querySelector('#butt');
			var root = host.createShadowRoot();
			root.textContent = 'Темная сторона';

Все части вместе

Создаем my-elem.html

			<template id="temp">
			    <div>Содержимое компоненты. <content></content></div>
			</template>
		
			var temp = document.querySelector('#temp');
			var proto = Object.create(HTMLElement.prototype);
			proto.createdCallback = function() {
			    var shadow = this.createShadowRoot();
			    shadow.appendChild(temp.content.cloneNode(true));
			};
			document.registerElement('my-elem', {prototype: proto});

Использование

			<link rel="import" href="my-elem.html">
			<my-elem>Содержимое одного элемента.</my-elem>
			<my-elem><b>Содержимое второго элемента.</b></my-elem>

  • Сложное разделяется на простые части
  • Улучшается семантика
  • Легко поддерживать / дорабатывать
  • Легко тестировать
  • Легко переиспользовать
  • component.kitchen и customelements.io

Базовый синтаксический сахар

			Polymer({
			    is: 'my-elem',
			    created: function() {
			        this.innerHTML = 'Привет из веб-компоненты!';
			    }
			});
		
			<my-elem></my-elem>

Наследование от нативных элементов

			Polymer({
			    is: 'my-elem',
			    extends: 'input'
			});
		
			<my-elem></my-elem>

Конвертация атрибутов к свойствам

			Polymer({
			    is: 'my-elem',
			    properties: {
			        userName: String,
			    }
			});
		
			<my-elem user-name="Denis"></my-elem>

Расширеная работа с шаблонами

			<dom-module id="my-elem">
			    <template>Привет из веб-компоненты!</template>
			    <content select=".myclass"></content>
			</dom-module>
		
			Polymer({
			    is: 'my-elem'
			});

Изолирование оформления (scoped css)

			<dom-module id="my-elem">
			    <style>/* Оформление */</style>
			    <template>Привет из веб-компоненты!</template>
			</dom-module>
		
			Polymer({
			    is: 'my-elem'
			});

Удобную работу с обработкой событий

			Polymer({
			    is: 'my-elem',
			    listeners: {
			        'click': 'handleClick'
			    },
			    handleClick: function(e) {
			        alert("Спасибо за клик!");
			} });

Установка обработчика через атрибуты

			<dom-module id="my-elem"><template>
			    <button on-click="handleClick">Жми!</button>
			</template></dom-module>
		
			Polymer({
			    is: 'my-elem',
			    handleClick: function(e) {
			        alert("Спасибо за клик!");
			} });

Отслеживание изменения свойств

			Polymer({
			    is: 'my-elem',
			    properties: { preload: Boolean, src: String },
			    observers: {
			        'preload src': 'updateImage'
			    },
			    updateImage: function( preload, src ){ что-то делаем }
			});

Декларативный дата биндинг

			<dom-module id="my-elem"><template>
			    <p>Привет! Меня зовут <span>{{user.name}}</span></p>
			</template></dom-module>
		
			Polymer({
			    is: 'my-elem',
			    properties: { user: { type: Object,
			                           value: { name: 'Денис' }
			} } });
